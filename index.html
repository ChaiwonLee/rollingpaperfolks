        </div>

        <div class="swiper-button-next"></div>

        <div class="swiper-button-prev"></div>

    </div>


    <script src="https://cdn.jsdelivr.net/npm/swiper@10/swiper-bundle.min.js"></script>

    <script>

        // ★ 본인의 구글 시트 CSV 링크를 아래 따옴표 안에 넣으세요!

        const CSV_URL = 'https://docs.google.com/spreadsheets/d/1gmE3hEl1nNZougcoSYIlVlW57NoQ_GRoAu6rljkE7Dg/edit?usp=sharing';


        async function loadMessages() {

            try {

                const response = await fetch(CSV_URL);

                const data = await response.text();

                const rows = data.split('\n').slice(1); // 첫 번째 줄(제목) 제외

                

                const wrapper = document.getElementById('paper-content');

                wrapper.innerHTML = ''; // 로딩 문구 제거


                rows.forEach(row => {

                    // 구글 시트의 열 순서에 맞춰 수정 (보통 타임스탬프, 이름, 메시지 순)

                    const columns = row.split(',');

                    if(columns.length >= 2) {

                        const name = columns[1];    // 2번째 칸: 이름

                        const message = columns[2]; // 3번째 칸: 메시지


                        const slide = `

                            <div class="swiper-slide">

                                <div class="message">" ${message} "</div>

                                <div class="author">- ${name} 학생 -</div>

                            </div>

                        `;

                        wrapper.innerHTML += slide;

                    }

                });


                // 슬라이더 초기화

                new Swiper('.swiper', {

                    navigation: { nextEl: '.swiper-button-next', prevEl: '.swiper-button-prev' },

                    loop: true,

                });


            } catch (error) {

                console.error('데이터 로드 실패:', error);

                document.getElementById('paper-content').innerHTML = '데이터를 가져오지 못했습니다.';

            }

        }


        loadMessages();

    </script>
